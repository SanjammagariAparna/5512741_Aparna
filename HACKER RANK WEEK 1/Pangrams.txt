char* pangrams(char* s) {
    int seen[26] = {0}; // track letters a-z
    int count = 0;

    for (int i = 0; s[i] != '\0'; i++) {
        char c = tolower(s[i]);
        if (c >= 'a' && c <= 'z') {
            int index = c - 'a';
            if (seen[index] == 0) {
                seen[index] = 1;
                count++;
            }
        }
    }

    if (count == 26) {
        return "pangram";
    } else {
        return "not pangram";
    }

}
