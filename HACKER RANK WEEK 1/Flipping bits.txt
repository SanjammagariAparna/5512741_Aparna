int main()
{
    FILE* fptr = fopen(getenv("OUTPUT_PATH"), "w");

    int q = parse_int(ltrim(rtrim(readline())));

    for (int q_itr = 0; q_itr < q; q_itr++) {
        long n = parse_long(ltrim(rtrim(readline())));

        long result = flippingBits(n);

        fprintf(fptr, "%ld\n", result);
    }

    fclose(fptr);

    return 0;
}
