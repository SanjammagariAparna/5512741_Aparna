#include stdio.h
#include stdlib.h

int cmpfunc(const void a, const void b) {
    return ((int)a - (int)b);
}

void findZigZagSequence(int arr[], int n) {
    qsort(arr, n, sizeof(int), cmpfunc);

    int mid = (n - 1)  2;  middle index
    int temp = arr[mid];
    arr[mid] = arr[n - 1];
    arr[n - 1] = temp;

     Reverse the second half
    int st = mid + 1;
    int ed = n - 2;  last element already swapped
    while (st = ed) {
        temp = arr[st];
        arr[st] = arr[ed];
        arr[ed] = temp;
        st++;
        ed--;
    }


    for (int i = 0; i  n; i++) {
        if (i  0) printf( );
        printf(%d, arr[i]);
    }
    printf(n);
}
