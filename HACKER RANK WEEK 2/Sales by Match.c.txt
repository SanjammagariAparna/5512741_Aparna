int sockMerchant(int n, int ar_count, int* ar) {
    int count[101] = {0};  // since 1 <= ar[i] <= 100
    int pairs = 0;

    for (int i = 0; i < ar_count; i++) {
        count[ar[i]]++;
    }

    for (int i = 1; i <= 100; i++) {
        pairs += count[i] / 2;  // each pair requires 2 socks
    }

    return pairs;
}
