
int digitSum(long long x) {
    int sum = 0;
    while (x > 0) {
        sum += x % 10;
        x /= 10;
    }
    return sum;
}

int superDigit(char* n, int k) {
    // Step 1: compute digit sum of n
    long long sum = 0;
    for (int i = 0; n[i] != '\0'; i++) {
        sum += (n[i] - '0');
    }

    // Step 2: multiply by k
    sum *= k;

    // Step 3: reduce to single digit
    while (sum >= 10) {
        sum = digitSum(sum);
    }

    return (int)sum;
}
